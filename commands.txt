git config --global user.name "username" [to add ur user name]

git config --global user.email "email" [to add ur user email]

git init [to initialize an empty git repository]

git status [to check to untracked files]

git add -A[to add all untracked files to staging area]

git commit -m "commit message" [to move staged files to local repository]

git config --list

git branch [list all branches]

git remote add remote_name "path" [to add remote repository to local]

git path remote_name branch_name [push ]

git remote -v [list all remotes]
